<style>
*{margin:0;padding:0;}
ul{ list-style-type:none;}
#div{width:360px; height:108px; background:green; margin:100px auto; position:relative; overflow:hidden;}
#div ul{position:absolute; left:0; top:0;}
#div ul li{ float:left; width:90px; height:108px;}
</style>
<script>
window.onload=function()
{
	var oDiv=document.getElementById('div')
	var oUl=oDiv.getElementsByTagName('ul')[0];
	var aLi=oDiv.getElementsByTagName('li');
	var timer=null;
	var speed=2;
	oUl.innerHTML=oUl.innerHTML+oUl.innerHTML;//把ul自己的html相加到了一起，两个加一块就是8个li
	oUl.style.width=aLi[0].offsetWidth*aLi.length+'px';//让ul的宽度等于随便哪个li(aLi[0].offsetWidth)的宽度然后*li的个数(aLi.length)再加上px
	function Move(){
		/*如果ul的向左滚动的offsetLeft值小于ul宽度offsetWidth的一半*/
		if(oUl.offsetLeft<-oUl.offsetWidth/2)  
		{
			oUl.style.left='0';//当滚到一半的时候然后再迅速把它拉过来！
		}
		if(oUl.offsetLeft>0)
		{
			oUl.style.left=-oUl.offsetWidth/2+'px';
		}
		oUl.style.left=oUl.offsetLeft+speed+'px';	//让ul的left不断的减小
	}
	timer=setInterval(Move,30)
	
	oDiv.onmouseover=function()
	{
		clearInterval(timer);
	}
	oDiv.onmouseout=function()
	{
	  timer=setInterval(Move,30)
	}
    document.getElementsByTagName('a')[0].onclick=function()
	{
		speed=-2
	}
    document.getElementsByTagName('a')[1].onclick=function()
	{
		speed=2
	}
	
}
</script>
<title>无标题文档</title>
</head>
